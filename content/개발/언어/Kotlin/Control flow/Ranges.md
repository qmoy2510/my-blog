package org.example.ControlFlow  
  
fun main(){

    //Kotlin에서 범위 설정할때는 보통 ..연산자를 이용함  
    //1..4는 1,2,3,4를 의미함 총 네번 반복 하는거

    //1..<4는 4를 포함하지 않는 수임  
    //고로 1,2,3  
    //거꾸로 반복하려면 downTo를 이용하면 됨  
    //4 downTo 1은 4,3,2,1임  
  
    //범위가 1씩 증가하는게 아니라 원하는 수 만큼 증가 시키고싶으면 Step을 쓰면 됨  
    //1..5 step 2 는 1,3,5임  
    //1에서 2씩 증가해서 5까지 하기 때문에  
    //1..<5 step 2는 1,3인거  
  
    //정수뿐만 아닌 char에서도 사용 가능함
